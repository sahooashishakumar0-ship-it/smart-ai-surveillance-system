<script>
async function loadAlerts() {
    try {
        const response = await fetch("http://127.0.0.1:5000/alerts");
        const data = await response.json();

        const table = document.getElementById("alertTable");
        table.innerHTML = "";

        if (data.length === 0) {
            table.innerHTML = "<tr><td colspan='4'>No alerts yet</td></tr>";
            return;
        }

        data.forEach(row => {
            const tr = document.createElement("tr");
            const eventClass = row[2].includes("DANGEROUS") ? "DANGEROUS" : "TRESPASSING";
            tr.innerHTML = `
                <td>${row[0]}</td>
                <td>${row[1]}</td>
                <td class="${eventClass}">${row[2]}</td>
                <td>${row[3]}</td>
            `;
            table.appendChild(tr);
        });

    } catch (error) {
        console.error("Error loading alerts:", error);
    }
}

// Load every 3 seconds
setInterval(loadAlerts, 3000);
loadAlerts();
</script>
